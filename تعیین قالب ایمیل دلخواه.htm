قالب ایمیل دلخواه موقع تعیین رمز عبور جدید

برای این کار باید از Notification استفاده کنیم که مراحل کار بصورت زیر است:

1- با دستور زیر یک Notification ایجاد میکنیم:

php artisan make:notification MailResetPassword

2- در مسیر App\Notifications، به فایل MailResetPassword.php میرویم و تغییرات لازم را ایجاد میکنیم که در نهایت کلاس مورد نظر به شکل زیر خواهد بود:

<?php

namespace App\Notifications;

use Illuminate\Bus\Queueable;
use Illuminate\Notifications\Notification;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Notifications\Messages\MailMessage;

class MailResetPassword extends Notification
{
    use Queueable;

    public $token;

    /**
     * Create a new notification instance.
     *
     * @return void
     */
    public function __construct($token)
    {
        $this->token = $token;
    }

    /**
     * Get the notification's delivery channels.
     *
     * @param  mixed  $notifiable
     * @return array
     */
    public function via($notifiable)
    {
        return ['mail'];
    }

    /**
     * Get the mail representation of the notification.
     *
     * @param  mixed  $notifiable
     * @return \Illuminate\Notifications\Messages\MailMessage
     */
    public function toMail($notifiable)
    {
        return (new MailMessage)
                    ->subject('تغییر رمز عبور')
                    ->greeting('سلام کاربر گرامی!')
                    ->line('اگر شما رمز عبورتان را فراموش کرده اید، رمز عبور جدیدی برای خودتان تعیین کنید!')
                    ->action('ورود به تعیین رمز عبور جدید', url(config('app.url').route('password.reset', $this->token, false)))
                    ->line('متشکریم!');
    }

    /**
     * Get the array representation of the notification.
     *
     * @param  mixed  $notifiable
     * @return array
     */
    public function toArray($notifiable)
    {
        return [
            //
        ];
    }
}

3- متد زیر را به مدل User اضافه میکنیم:

/**
 * Send a password reset email to the user
 */
public function sendPasswordResetNotification($token)
{
    $this->notify(new MailResetPassword($token));
}

4- مسیر Notification که ساخته ایم را برای مدل User، یوز میکنیم:

/**
 * Send a password reset email to the user
 */
public function sendPasswordResetNotification($token)
{
    $this->notify(new MailResetPasswordToken($token));
}

5- در خط فرمان دستور زیر را اجرا میکنیم:

php artisan vendor:publish

6- برای تغییر قالب ایمیل به مسیر زیر میرویم:

/resources/views/vendor/notifications

7- فایل email.blade.php را بصورت زیر تغییر میدهیم:

@component('mail::message')

{{-- Greeting --}}
@if (! empty($greeting))
    <h2>{{ $greeting }}</h2>
@else
    @if ($level == 'error')
    # Whoops!
    @else
    <h2>سلام!</h2>
    @endif
@endif


{{-- Intro Lines --}}
@foreach ($introLines as $line)
    {{ $line }}
@endforeach


{{-- Action Button --}}
@isset($actionText)
<?php
    switch ($level) {
        case 'success':
            $color = 'green';
            break;
        case 'error':
            $color = 'red';
            break;
        default:
            $color = 'blue';
    }
?>
@component('mail::button', ['url' => $actionUrl, 'color' => $color])
{{ $actionText }}
@endcomponent
@endisset

{{-- Outro Lines --}}
@foreach ($outroLines as $line)
    {{ $line }}
@endforeach

<br>
وب سایت {{ config('app.name') }}


@endcomponent
