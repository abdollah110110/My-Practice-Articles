تابع func_get_args

func_num_args(): تعداد آرگومانهای ارسال شده به تابع را برمیگرداند.

func_get_arg(0): آرگومان مشخص شده بر اساس شماره را برمیگرداند.

func_get_args(): همه آرگومانهای ارسال شده را تحت یک ارایه برمیگرداند.

مثال:


<?php 

function imploadDate(){

    $number = func_num_args();

    echo 'Number of arguments: ' . $number . '<br /><br />' . PHP_EOL;

    

    $argArray = [];

    if($number <= 1 && is_array(func_get_arg(0))){

        echo '<pre>' . print_r(func_get_arg(0), true) . '' . PHP_EOL;

        foreach(func_get_arg(0) as $value){

            $argArray[] = $value;

        }

    }

    else{

        echo '<pre>' . print_r(func_get_args(), true) . '' . PHP_EOL;

        $argArray = func_get_args();

    }

    

    $date = implode('/', $argArray);

    echo 'Date: ' . $date . '<br /><br />' . PHP_EOL;

}



$year = '2020';

$month = '06';

$day = '14';



$arrayDate = ['year' => '2020', 'month' => '06' , 'day' => '14'];



echo 'Arg is array: ' . PHP_EOL;

imploadDate($arrayDate);



echo 'Args are variables: ' . PHP_EOL;

imploadDate($year, $month, $day);

?>
