سطح بندی کاربران در laravel

برای سطح بندی کاربران باید چهار جدول ایجاد کنیم که در خط فرمان دستور مایگریشن هر کدام را بصورت زیر وارد میکنیم:

php artisan make:migration create_role_table --create=role

php artisan make:migration create_permission_table --create=permission

php artisan make:migration create_permission_role_table --create=permission_role

php artisan make:migration create_role_user_table --create=role_user

 

برای ایجاد و حذف جدول roles، متدهای زیر را در فایل مایگریشن مربوط به آن مینویسیم:

public function up()
{
    Schema::create('roles', function (Blueprint $table) {
        $table->increments('id');
        $table->string('name');
        $table->string('label')->nullable();
        $table->timestamps();
    });

}

public function down()
{
    Schema::dropIfExists('roles');
}

 

برای ایجاد و حذف جدول permissions، متدهای زیر را در فایل مایگریشن مربوط به آن مینویسیم:

public function up()
{
    Schema::create('permissions', function (Blueprint $table) {
        $table->increments('id');
        $table->string('name');
        $table->string('label')->nullable();
        $table->timestamps();
    });
}

public function down()
{
    Schema::dropIfExists('permissions');
}

 

برای ایجاد رابطه چند به چند جدولهای roles و permissions، متدهای زیر را در فایل مایگریشن مربوط به آن مینویسیم:

public function up()
{
    Schema::create('permission_role', function (Blueprint $table) {

        $table->integer('role_id')->unsigned();
        $table->foreign('role_id')->references('id')->on('roles')->onDelete('cascade');

        $table->integer('permission_id')->unsigned();
        $table->foreign('permission_id')->references('id')->on('permissions')->onDelete('cascade');

        $table->primary(['role_id', 'permission_id']);
    });
}

public function down()
{
    Schema::dropIfExists('permission_role');
}

 

برای ایجاد رابطه چند به چند جدولهای roles و users، متدهای زیر را در فایل مایگریشن مربوط به آن مینویسیم:

public function up()
{
    Schema::create('role_user', function (Blueprint $table) {

        $table->integer('role_id')->unsigned();
        $table->foreign('role_id')->references('id')->on('roles')->onDelete('cascade');

        $table->integer('user_id')->unsigned();
        $table->foreign('user_id')->references('id')->on('users')->onDelete('cascade');

        $table->primary(['role_id', 'user_id']);
    });
}

public function down()
{
    Schema::dropIfExists('role_user');
}

 

با وارد کردن دستورات زیر در خط فرمان، مدلهای Role و Permission را ایجاد میکنیم:

php artisan make:model Role

php artisan make:model Permission

 

در این مرحله برای ایجاد روابط بین دو جدول users و roles، دو متد زیر را به ترتیب در مدلهای User و Role وارد میکنیم:

// User Model
public function roles()
{
    return $this->belongsToMany(Role::class);
}


// Role Model
protected $fillable = ['name', 'label'];

public function users(){
    return $this->belongsToMany(User::class);
}

 

و بعد برای ایجاد روابط بین دو جدول permissions و roles، دو متد زیر را به ترتیب در مدلهای Role و Permission وارد میکنیم:

// Role Model
public function permissions()
{
    return $this->belongsToMany(Permission::class);
}

// Permission Model
protected $fillable = ['name', 'label'];

public function roles()
{
    return $this->belongsToMany(Role::class);
}

استفاده از role ها و permission ها برای سطح بندی

1- در مدل User متد hasRole را اضافه میکنیم:

public function hasRole($role)
{
    if(is_string($role)){
        return $this->roles->contains('name', $role);
    }

    return !!$role->intersect($this->roles)->count();
}


2- در فایل AuthServiceProvider و در متد boot کد زیر و متد getPermissions رو اضافه میکنیم:

public function boot()
{
    $this->registerPolicies();

    foreach($this->getPermissions() as $permission)
    {
        Gate::define($permission->name, function($user) use($permission) {
            return $user->hasRole($permission->roles);
        });
    }
}

protected function getPermissions(){
    return Permission::with('roles')->get();
}

 

3- حالا در یک ویوی خاص مثلا index.blade.php وجود پرمیشن edit-article رو بصورت زیر با استفاده از can بررسی میکنیم:

@can('edit-article')
<a href="{{ route('admin_article.edit', ['article' => $article->slug]) }}" class="btn btn-sm btn-warning btn-block">
    <i class="glyphicon glyphicon-pencil"></i>
</a>
@endcan

یعنی کاربر وقتی میتونه به لینک edit دسترسی داشته باشه که پرمیشن edit-article رو داشته باشه.

میتوانیم در کنترلرها بصورت زیر از متد hasRole استفاده کنیم:

auth()->loginUsingId(1);

dd(auth()->user()->hasRole('manager'));
// true

// ارسال آبجکت
dd(auth()->user()->hasRole(Permission::whereName('edit-article')->first()->roles));
// true

 

 

کنترلرهای مورد نیاز برای انجام CRUD

برای انجام عملیات CRUD، متدهای مورد نیاز را در کنترلرهای زیر به ترتیب ایجاد میکنیم:

1- کنترلر PermissionController:

<?php

namespace App\Http\Controllers\Admin;

use App\Permission;
use Illuminate\Http\Request;
use App\Http\Controllers\Controller;

class PermissionController extends Controller
{
    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function index()
    {
        $permissions = Permission::latest()->paginate(50);

        return view('admin.permissions.index', compact('permissions'));
    }

    /**
     * Show the form for creating a new resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function create()
    {
        return view('admin.permissions.create');
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(Request $request)
    {
        $this->validate($request,[
            'name' => 'required|Max:100',
            'label' => 'required|Max:100',
        ]);

        Permission::create([
            'name' => $request->name,
            'label' => $request->label,
        ]);

        session()->flash('message', 'مجوز ' . $request->label . ' با موفقیت ایجاد شد.');

        return redirect('admin/permissions/index');
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Permission  $permission
     * @return \Illuminate\Http\Response
     */
    public function show(Permission $permission)
    {
        return view('admin.permissions.show', compact('permission'));
    }

    /**
     * Show the form for editing the specified resource.
     *
     * @param  \App\Permission  $permission
     * @return \Illuminate\Http\Response
     */
    public function edit(Permission $permission)
    {
        return view('admin.permissions.edit', compact('permission'));
    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Permission  $permission
     * @return \Illuminate\Http\Response
     */
    public function update(Request $request, Permission $permission)
    {
        $this->validate($request,[
            'name' => 'required|Max:100',
            'label' => 'required|Max:100',
        ]);

        $permission->name = $request->name;
        $permission->label = $request->label;
        $permission->save();

        session()->flash('message', 'مجوز ' . $permission->label . ' با موفقیت ویرایش شد.');

        return redirect('admin/permissions/index');
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Permission  $permission
     * @return \Illuminate\Http\Response
     */
    public function destroy(Permission $permission)
    {
		$permission->roles()->detach();
		
        $permission->delete();

        session()->flash('message', 'مجوز ' . $permission->label . ' با موفقیت حذف شد.');

        return redirect('admin/permissions/index');
    }
}

 

2- کنترلر RoleController:

<?php

namespace App\Http\Controllers\Admin;

use App\Role;
use App\Permission;
use Illuminate\Http\Request;
use App\Http\Controllers\Controller;

class RoleController extends Controller
{
    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function index()
    {
        $roles = Role::latest()->paginate(50);

        return view('admin.roles.index', compact('roles'));
    }

    /**
     * Show the form for creating a new resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function create()
    {
        $permissions = Permission::all();

        return view('admin.roles.create', compact('permissions'));
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(Request $request)
    {
        $this->validate($request, [
            'name' => 'required|Max:100',
            'label' => 'required|Max:100',
            'permissions_id' => 'required',
        ]);

        $role = Role::create([
            'name' => $request->name,
            'label' => $request->label,
        ]);

        $role->permissions()->sync($request->input('permissions_id'));

        session()->flash('message', 'مقام ' . $request->label . ' با موفقیت ایجاد شد.');

        return redirect('admin/roles/index');
    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Role  $role
     * @return \Illuminate\Http\Response
     */
    public function show(Role $role)
    {
        return view('admin.roles.show', compact('role'));
    }

    /**
     * Show the form for editing the specified resource.
     *
     * @param  \App\Role  $role
     * @return \Illuminate\Http\Response
     */
    public function edit(Role $role)
    {
        $permissions = Permission::all();

        return view('admin.roles.edit', compact('role', 'permissions'));
    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Role  $role
     * @return \Illuminate\Http\Response
     */
    public function update(Request $request, Role $role)
    {
        $this->validate($request, [
            'name' => 'required|Max:100',
            'label' => 'required|Max:100',
            'permissions_id' => 'required',
        ]);

        $role->name = $request->name;
        $role->label = $request->label;
        $role->save();

        $role->permissions()->sync($request->input('permissions_id'));

        session()->flash('message', 'مقام ' . $request->label . ' با موفقیت ویرایش شد.');

        return redirect('admin/roles/index');
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Role  $role
     * @return \Illuminate\Http\Response
     */
    public function destroy(Role $role)
    {
        $role->permissions()->detach();
		$role->users()->detach();
		
		$role->delete();

        session()->flash('message', 'مقام ' . $role->label . ' با موفقیت حذف شد.');

        return redirect('admin/roles/index');
    }
}

permissions_id آرایه ای شامل آیدیهای مجوزها میباشد.

اعمال سطوح دسترسی به route ها و view ها

 

برای این کار کافی است در route مورد نظر middleware را وارد کنیم یا route های مورد نظر را در یک Route::group قرار میدهیم:

Route::get('/user/index', 'UserController@index')->middleware('can:manager')->name('admin_user.index');

Route::group(
    [
        'middleware' => 'can:edit-users',
    ],
    function(){
        Route::get('/user/index', 'UserController@index')->name('admin_user.index');
        Route::get('/user/show/{user}', 'UserController@show')->name('admin-user.show');
        Route::get('/user/edit/{user}', 'UserController@edit')->name('admin-user.edit');
        Route::put('/user/update/{user}', 'UserController@update')->name('admin-user.update');
        Route::delete('/user/destroy/{user}', 'UserController@destroy')->name('admin-user.destroy');
    }
);

یا دستور زیر را در کنترلر مورد نظر وارد میکنیم:

public function index()
{
    $this->authorize('edit-users');

// ...
